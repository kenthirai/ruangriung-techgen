```sql
-- Users table (optional, if you want to track users beyond session)
CREATE TABLE users (
  id TEXT PRIMARY KEY,
  github_username TEXT,
  email TEXT,
  avatar_url TEXT,
  created_at INTEGER DEFAULT (unixepoch())
);

-- Sessions (for anonymous usage tracking)
CREATE TABLE sessions (
  id TEXT PRIMARY KEY,
  created_at INTEGER DEFAULT (unixepoch()),
  last_active INTEGER DEFAULT (unixepoch())
);

-- Generations history
CREATE TABLE generations (
  id TEXT PRIMARY KEY,
  session_id TEXT NOT NULL,
  user_id TEXT,  -- nullable if using sessions only
  type TEXT NOT NULL, -- 'text' | 'image' | 'video' | 'audio' | 'embedding'
  model TEXT NOT NULL,
  prompt TEXT NOT NULL,
  negative_prompt TEXT,
  result_url TEXT,
  result_data TEXT, -- JSON blob for structured responses (chat, embeddings)
  params TEXT, -- JSON of all generation params
  cost REAL, -- pollen cost
  is_favorite INTEGER DEFAULT 0,
  created_at INTEGER DEFAULT (unixepoch()),
  FOREIGN KEY (session_id) REFERENCES sessions(id),
  FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Favorites (alternative to is_favorite flag)
CREATE TABLE favorites (
  id TEXT PRIMARY KEY,
  generation_id TEXT NOT NULL,
  session_id TEXT NOT NULL,
  created_at INTEGER DEFAULT (unixepoch()),
  FOREIGN KEY (generation_id) REFERENCES generations(id),
  FOREIGN KEY (session_id) REFERENCES sessions(id)
);

-- Models cache (sync from API periodically)
CREATE TABLE models_cache (
  id TEXT PRIMARY KEY,
  type TEXT NOT NULL, -- 'text' | 'image' | 'video' | 'audio' | 'embedding'
  name TEXT NOT NULL,
  capabilities TEXT, -- JSON
  pricing TEXT, -- JSON
  metadata TEXT, -- JSON (context_window, modalities, voices, etc.)
  updated_at INTEGER DEFAULT (unixepoch())
);

-- User API keys (if you're managing keys on behalf of users)
CREATE TABLE user_api_keys (
  id TEXT PRIMARY KEY,
  user_id TEXT NOT NULL,
  key_prefix TEXT NOT NULL, -- 'sk_' or 'pk_'
  key_hash TEXT NOT NULL, -- hashed for security
  name TEXT,
  redirect_uris TEXT, -- JSON array
  earnings_enabled INTEGER DEFAULT 0,
  created_at INTEGER DEFAULT (unixepoch()),
  expires_at INTEGER,
  FOREIGN KEY (user_id) REFERENCES users(id)
);

-- Indexes
CREATE INDEX idx_generations_session ON generations(session_id);
CREATE INDEX idx_generations_user ON generations(user_id);
CREATE INDEX idx_generations_type ON generations(type);
CREATE INDEX idx_generations_created ON generations(created_at DESC);
CREATE INDEX idx_favorites_session ON favorites(session_id);
```

### рџ”§ Cloudflare Worker (API Backend)

The Worker handles:
1. **Auth proxy** вЂ” forward BYOP auth, store user sessions
2. **Generation history** вЂ” CRUD operations on D1
3. **Model cache** вЂ” periodically sync models from Pollinations API
4. **Rate limiting** вЂ” per IP/session (optional)
5. **CORS** вЂ” for frontend

**Hono.js example structure:**
```typescript
// workers/api/index.ts
import { Hono } from 'hono'
import { cors } from 'hono/cors'
import { authRoutes } from './routes/auth'
import { generationRoutes } from './routes/generations'
import { historyRoutes } from './routes/history'
import { modelRoutes } from './routes/models'

const app = new Hono()

app.use('*', cors())
app.route('/api/auth', authRoutes)
app.route('/api/generations', generationRoutes)
app.route('/api/history', historyRoutes)
app.route('/api/models', modelRoutes)

export default app
```

### рџ”ђ Authentication Flow Implementation

**Web Redirect Flow:**

```typescript
const clientId = import.meta.env.VITE_POLLINATIONS_PK_KEY

  // Redirect to dashboard
```

**Device Flow (for headless/CLI):**
```typescript
// 1. Request device code
const { device_code, user_code, verification_uri } = await fetch(
  'https://enter.pollinations.ai/api/device/code',
  { method: 'POST', body: JSON.stringify({ client_id: 'pk_...', scope: 'generate' }) }
)
// Show user_code to user with instructions

// 2. Poll for token
const poll = async () => {
  const res = await fetch('https://enter.pollinations.ai/api/device/token', {
    method: 'POST',
    body: JSON.stringify({ device_code })
  })
  const data = await res.json()
  if (data.access_token) {
    localStorage.setItem('pollinations_api_key', data.access_token)
    return data.access_token
  }
  if (data.error === 'authorization_pending') {
    await sleep(5000)
    return poll()
  }
  throw new Error(data.error)
}
```
